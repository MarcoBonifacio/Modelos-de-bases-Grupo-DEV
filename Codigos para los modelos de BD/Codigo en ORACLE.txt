EJEMPLO 1
CREATE TABLE Cliente (
    idC NUMBER PRIMARY KEY,
    tipoDocumC VARCHAR2(20),
    nroDocumC VARCHAR2(20),
    nombresC VARCHAR2(50),
    paternoC VARCHAR2(50),
    maternoC VARCHAR2(50),
    celularC VARCHAR2(15)
);

CREATE TABLE Paquete (
    idP NUMBER PRIMARY KEY,
    descripcionP VARCHAR2(100),
    pesoP NUMBER(10, 2)
);

CREATE TABLE Comprobante (
    idComp NUMBER PRIMARY KEY,
    fechaHoraComp TIMESTAMP,
    totalComp NUMBER(10, 2),
    OrigenComp VARCHAR2(50),
    DestinoComp VARCHAR2(50),
    idC NUMBER,
    CONSTRAINT fk_shalom_cliente FOREIGN KEY (idC) REFERENCES Cliente(idC)
);

CREATE TABLE DetalleDelComprobante (
    idDet NUMBER PRIMARY KEY,
    precioPorKiloDet NUMBER(10, 2),
    subtotalDet NUMBER(10, 2),
    idComp NUMBER,
    idP NUMBER,
    CONSTRAINT fk_shalom_comp FOREIGN KEY (idComp) REFERENCES Comprobante(idComp),
    CONSTRAINT fk_shalom_paq FOREIGN KEY (idP) REFERENCES Paquete(idP)
);

CREATE TABLE Empleado (
    idE NUMBER PRIMARY KEY,
    tipoDocumE VARCHAR2(20),
    nroDocumE VARCHAR2(20),
    nonbresE VARCHAR2(50),
    paternoE VARCHAR2(50),
    maternoE VARCHAR2(50),
    celularE VARCHAR2(15)
);

CREATE TABLE Cuenta (
    idCuenta NUMBER PRIMARY KEY,
    nroCuenta VARCHAR2(50),
    saldoCuenta NUMBER(18, 2),
    idB NUMBER,
    idC NUMBER,
    CONSTRAINT fk_giros_banco FOREIGN KEY (idB) REFERENCES Banco(idB),
    CONSTRAINT fk_giros_cliente FOREIGN KEY (idC) REFERENCES Cliente(idC)
);

CREATE TABLE Operacion (
    idOpe NUMBER PRIMARY KEY,
    tipoOpe VARCHAR2(50),
    fechaHoraOpe TIMESTAMP,
    montoOpe NUMBER(18, 2),
    comisionOpe NUMBER(10, 2),
    idCuenta NUMBER,
    idE NUMBER,
    CONSTRAINT fk_giros_cuenta FOREIGN KEY (idCuenta) REFERENCES Cuenta(idCuenta),
    CONSTRAINT fk_giros_emp FOREIGN KEY (idE) REFERENCES Empleado(idE)
);
